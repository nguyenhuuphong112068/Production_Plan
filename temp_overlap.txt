checkTimeOverlapForEmployee(personId, $item);
                    if (check.overlap) {
                        hasOverlap = true;
                        overlapMsg = check.message;
                        return false;
                    }
                }
            });

            if (hasOverlap) {
                Swal.fire({
                    icon: 'error',
                    title: 'Trùng lịch làm việc',
                    text: overlapMsg
                });
                isProgrammaticChange = true;
                $el.val(prevVal);
                isProgrammaticChange = false;
            } else {
                $el.data('prev-val', $el.val());
                updateTimelines();
                updateSidebarPersonnelTimes();
            }
        });
